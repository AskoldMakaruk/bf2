# Граматика мови

## Коментар

```text
коментар -> блок_коментар | рядковий_коментар | дефіс_коментар ;
блок_коментар -> "/*" (коментар | символ) * "*/" ;
рядковий_коментар -> "//" * ;
дефіс_коментар -> "--" *;
```
# Програма 

```text
програма -> декларація* EOF;
декларація -> декларація_змінної | твердження | декларація_функції | декларація_типу ;
декларація_змінної -> ІДЕНТИФІКАТОР ("це" тип_параметри)? ("є" вираз)? "." ;
декларація_типу -> "тип" (ім'я|символ) "це" тип_параметри? "є" тип_параметри? "." ;
```
## Твердження
```text
твердження -> вираз_твердження | прінт_вираз | блок ;
блок -> "{" декларація* "}" ;
вираз_твердження -> вираз "." ;
прінт_вираз -> "вивести" вираз "." ;
```
## Типи

```text
тип_параметри -> (тип|тип_параметр) ("," тип|тип_параметр)* ;
тип_параметр -> ім'я ;
тип -> ім'я | "нічо" | "я" | тип_масив | тип_адресат ;
тип_масив -> тип "[]" ;
тіло_типу -> поле_типу (сепар поле_типу) ".";
поле_типу -> тип ім'я ;
```

## Вирази

```text
вираз -> літерал | унарний | бінарний | групування | вираз_масив | якщо_вираз ;
присвоєння -> ІДЕНТИФІКАТОР "це" рівність ;
літерал -> ЧИСЛО | ТЕКСТ | "так" | "ні" | "нічо" | "(" вираз ")" ;
унарний -> ("-" | "не") вираз ;
бінарний -> вираз операція вираз ;
операція -> "==" | "!=" | "<" | "<=" | ">" | ">="
| "+" | "-" | "*" | "/" ;
вираз_масив -> "[" (вираз ("," вираз)*)? "]" ;
якщо_вираз -> "якщо" вираз "то" вираз (інакше якщо_вираз)* ("інакше" вираз)? ;
```

## Логічні вирази
    
```text
лог_або -> лог_та ( "або" лог_та )* ;
лог_та -> рівність ( "та" рівність )* ;
```
## Математичні вирази

```text
мат_вираз -> рівність ;
рівність -> порівняння ( ( "дорівнює" | "не дорівнює" ) порівняння )* ;
порівняння -> мат_дод ( ( "==" | "!=" | "<" | "<=" | ">" | ">=" ) мат_дод )* ;
мат_дод -> мат_множ ( ( "+" | "-" ) мат_множ )* ;
мат_множ ->  унарний ( ( "/" | "*" ) унарний )* ;
унарний -> ("-" | "не") унарний | основний ;
основний -> ЧИСЛО | ТЕКСТ | "так" | "ні" | "нічо" | "(" мат_вираз ")" ;


бінарний -> унарний операція унарний | основний операція основний ;
```

## Утиліти

```text
символ -> "#" ім'я ;
ім'я -> буква (буква|цифра)*;
буква -> "а".."я" | "А".."Я" | "_" | "'" ;
цифра ->  "0".."9" ;
сепар -> (",")? ("\n")? ; 
EOF -> "\0" ;
```
